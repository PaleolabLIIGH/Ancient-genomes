```
module load anaconda3/2021.05
source activate pyega3-5.2.0

###############################################
# Path to the credentials file
CREDENTIALS_FILE="/panfs/pan5/mavila/lcarrillo/biobank_mx/CREDENTIALS_FILE.json" #Replace with your credentials

# Output directory to save the downloaded files
OUTPUT_DIR="/panfs/pan5/mavila/lcarrillo/biobank_mx/genomes" #Modify your path

# Dataset identifier to download
DATASET_ID="EGAD00001008354"  # Replace with the ID of your dataset

# Ensure the output directory exists
mkdir -p $OUTPUT_DIR

# Run the pyEGA3 command to fetch the full dataset
pyega3 -cf $CREDENTIALS_FILE -c 10 fetch $DATASET_ID --output-dir $OUTPUT_DIR

exit
```

### 23.4 Modify the md5 files
To verify that all the files were downloaded correctly, it is necessary to restructure the files using the following loop:
```
cd /panfs/pan5/mavila/lcarrillo/biobank_mx #Go to your work directory
for file in genomes/EGAF0000571984*/MXB_*.md5; do #Modify accordingly the output directory and the patterns
     checksum=$(cat "$file")
     dir=$(dirname "$file")
     filename=$(basename "$file" .md5)
     echo "$checksum  $dir/$filename" > "$file"
done
```
