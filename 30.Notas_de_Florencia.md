import requests
import pandas as pd

#3# List of Sample Aliases to search 
sample_aliases = [
    "HBV_SID005", "HBV_SJN005", "HBV_SJN013", "HBV_SMH004", "HBV_STR144",
    "HBV_SUA002", "HBV_SUA007", "HBV_TRI011", "HBV_TTK002", "HBV_UOO011",
    "HBV_UOO024", "HBV_UOO033", "HBV_UZZ061", "HBV_UZZ075", "HBV_UZZ081",
    "HBV_UZZ099", "HBV_VAR021", "HBV_VLI016", "HBV_VLI060", "HBV_VTZ008",
    "HBV_VTZ011", "HBV_WEH008", "HBV_WEH009", "HBV_WSN010", "HBV_WSN012",
    "HBV_YUN048", "HBV_ZWE008", "HBV_G78T", "HBV_G79T", "HBV_G27T", "HBV_G83T"
]

#4# Function to remove “HBV_”
def clean_sample_name(sample):
    return sample.replace("HBV_", "") if sample.startswith("HBV_") else sample

#5# ENA API URL
base_url = "https://www.ebi.ac.uk/ena/portal/api/search"
params = {
    "result": "read_run",
    "query": "study_accession=PRJEB45699",
    "fields": "sample_alias,run_accession,fastq_ftp",
    "format": "json",
    "limit": 1000
}

#6# Request API
response = requests.get(base_url, params=params)
data = response.json()

#7# Extract data
filtered_data = []
for entry in data:
    original_sample = entry.get("sample_alias")
    if original_sample in sample_aliases:
        clean_sample = clean_sample_name(original_sample)
        fastq_url = entry.get("fastq_ftp", "").split(";")[0]  # Solo el primer link

        filtered_data.append({
            "Sample Alias": clean_sample,
            "Run Accession": entry.get("run_accession"),
            "FASTQ URL": fastq_url
        })

#8#  Create DataFrame and save to CSV
df = pd.DataFrame(filtered_data)
df.to_csv("ENA_FASTQ_Links.csv", index=False)

print("Archivo 'ENA_FASTQ_Links.csv' guardado con Sample Alias, Run Accession y FASTQ URL.")
