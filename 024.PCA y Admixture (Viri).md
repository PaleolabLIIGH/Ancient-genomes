module load htslib/1.2.1
module load samtools/1.2
module load r/3.2.1
module load mapdamage/2.0

file=$1

base=`basename $file .bam`

mapDamage -i $file -r $2 --rescale --rescale-out $base.rescaled.bam

samtools index $base.rescaled.bam

exit;
