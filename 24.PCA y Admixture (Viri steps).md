That BAM file is the one we want to rescale. We can use this script to search for BAM files in the root directory and automatically submit a qsub job for each subdirectory.
```
#!/bin/bash

# Root directory
ROOT_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38"

# Script to execute
MAPDAMAGE_SCRIPT="/mnt/Cromosoma/mavila/vvilla/scripts/mapdamage_v_realign.MD.sh"
REFERENCE_FILE="/panfs/pan5/mavila/vvilla/genome_ref/human/GRCh38.p13/GRCh38.p13_chr-only/GRCh38.p13.rCRS.fasta"

# Load necessary modules
module load htslib/1.2.1
module load samtools/1.2
module load r/3.2.1
module load mapdamage/2.0

# Change to the root directory
cd "$ROOT_DIR" || exit

# Iterate over each subdirectory with "-bam" suffix
for SUBDIR in *-bam/; do
    # Remove the trailing slash from the subdirectory name
    SUBDIR=${SUBDIR%/}

    # Change to the subdirectory
    cd "$SUBDIR" || continue

    # Find the BAM file with "realign.pmd3.MD.bam" suffix in the subdirectory
    BAMFILE=$(ls *realign.pmd3.MD.bam 2>/dev/null)

    # Check if a BAM file was found
    if [ -z "$BAMFILE" ]; then
        echo "No BAM file with suffix 'realign.MD.bam' found in $SUBDIR"
        cd ..
        continue
    fi

    # Execute the mapdamage script
    sh "$MAPDAMAGE_SCRIPT" "$BAMFILE" "$REFERENCE_FILE"

    # Return to the root directory
    cd ..

    # Optional: Wait for a short time before processing the next subdirectory
    # Uncomment the following line if needed
    sleep 60
done

exit
```

```
#!/bin/bash

# Root directory for BAM files
ROOT_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38"

# PCA analysis directory
PCA_DIR="$ROOT_DIR/PCA_analysis"

# Reference panel files
REFERENCE_PANEL_DIR="/panfs/pan5/mavila/vvilla/Reference_panels/1000GP_MEGAsites/hg38"
REFERENCE_FILES=(
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.bed"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.bim"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.fam"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.ped"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.map"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.rand.ped"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.rand.map"
    "1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno.pos"
)

# Ensure PCA directory exists
mkdir -p "$PCA_DIR"

# Change to the root directory
cd "$ROOT_DIR" || exit

# Iterate over each subdirectory with "-bam" suffix
for SUBDIR in *-bam/; do
    # Remove the trailing slash from the subdirectory name
    SUBDIR=${SUBDIR%/}

    # Change to the subdirectory
    cd "$SUBDIR" || continue

    # Find the BAM file and its index with "realign.MD.rescaled.bam" suffix
    BAMFILE=$(ls *realign.pmd3.MD.rescaled.bam 2>/dev/null)
    BAI_FILE="$BAMFILE.bai"

    # Check if the BAM and BAI files exist
    if [ -z "$BAMFILE" ] || [ ! -f "$BAI_FILE" ]; then
        echo "Required files not found in $SUBDIR"
        cd ..
        continue
    fi

    # Extract individual ID from the BAM file name (assuming prefix format like "001-H089")
    INDIVIDUAL_ID=$(echo "$BAMFILE" | cut -d'_' -f1)
    INDIVIDUAL_DIR="$PCA_DIR/${INDIVIDUAL_ID}_PCA"

    # Create individual PCA directory
    mkdir -p "$INDIVIDUAL_DIR"

    # Create symbolic links for BAM and BAI files
    ln -sf "$ROOT_DIR/$SUBDIR/$BAMFILE" "$INDIVIDUAL_DIR/$BAMFILE"
    ln -sf "$ROOT_DIR/$SUBDIR/$BAI_FILE" "$INDIVIDUAL_DIR/$BAI_FILE"

    # Create symbolic links for reference panel files
    for REF_FILE in "${REFERENCE_FILES[@]}"; do
        ln -sf "$REFERENCE_PANEL_DIR/$REF_FILE" "$INDIVIDUAL_DIR/$REF_FILE"
    done

    # Return to the root directory
    cd ..
done

exit
```

This is the script:
```
#!/bin/bash

# Root directory for PCA analysis
PCA_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/PCA_analysis"

# Script path for PCA processing
PCA_SCRIPT="/mnt/Cromosoma/mavila/backup/lcarrillo/scripts/make_evec_per_sample_hg38_lau_mD2_filters.Q20_v2.2.sh"

# Reference panel file
REFERENCE_FILE="1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno"

# Ensure PCA script exists
if [ ! -f "$PCA_SCRIPT" ]; then
    echo "PCA script not found: $PCA_SCRIPT"
    exit 1
fi

# Change to PCA directory
cd "$PCA_DIR" || exit

# Iterate over each individual directory
for DIR in *_PCA/; do
    # Define individual directory
    INDIVIDUAL_DIR="$PCA_DIR/$DIR"

    # Change to individual directory
    cd "$INDIVIDUAL_DIR" || continue

    # Find the BAM file with the expected suffix
    BAMFILE=$(ls *pmd3.trimmed9.MD.bam 2>/dev/null)

    # Check if BAM file exists
    if [ -z "$BAMFILE" ]; then
        echo "No BAM file found in $INDIVIDUAL_DIR"
        cd ..
        continue
    fi

    # Submit the job using qsub
    echo "Submitting job for $BAMFILE in $INDIVIDUAL_DIR"
    qsub "$PCA_SCRIPT" "$BAMFILE" "$REFERENCE_FILE"

    # Return to PCA directory
    cd ..
    sleep 60
done

exit
```
Modify the `PCA_DIR` and the suffix in `BAMFILE` if necessary.
