# Ensure PCA directory exists
mkdir -p "$PCA_DIR"

# Change to the trimmed BAM files directory
cd "$TRIMMED_DIR" || exit

# Iterate over each trimmed BAM file
for BAMFILE in *pmd3.trimmed9.MD.bam; do
    # Check if BAM file exists
    if [ ! -f "$BAMFILE" ]; then
        echo "BAM file not found: $BAMFILE"
        continue
    fi

    # Check if index file exists
    BAI_FILE="$BAMFILE.bai"
    if [ ! -f "$BAI_FILE" ]; then
        echo "Index file not found for: $BAMFILE"
        continue
    fi

    # Extract individual ID from the BAM file name (assuming prefix format like "001-H089")
    INDIVIDUAL_ID=$(echo "$BAMFILE" | cut -d'_' -f1)
    INDIVIDUAL_DIR="$PCA_DIR/${INDIVIDUAL_ID}_PCA"

    # Create individual PCA directory
    mkdir -p "$INDIVIDUAL_DIR"

    # Create symbolic links for BAM and BAI files
    ln -sf "$TRIMMED_DIR/$BAMFILE" "$INDIVIDUAL_DIR/$BAMFILE"
    ln -sf "$TRIMMED_DIR/$BAI_FILE" "$INDIVIDUAL_DIR/$BAI_FILE"

    # Create symbolic links for reference panel files
    for REF_FILE in "${REFERENCE_FILES[@]}"; do
        ln -sf "$REFERENCE_PANEL_DIR/$REF_FILE" "$INDIVIDUAL_DIR/$REF_FILE"
    done
done

exit

### 24.3 PCA by individual

Now that we have created a folder named `PCA_analysis` and, within it, subdirectories for each individual containing the necessary BAM and BAI files 
along with the symbolic links to the reference panel, we can begin the PCA analysis for each individual. To do so, copy this script into your folder:
```

cp /mnt/Cromosoma/mavila/backup/lcarrillo/scripts/make_pca_by_individual.sh /mnt/Cromosoma/mavila/lcarrillo/scripts #Modify destine
```
This is the script:
```
#!/bin/bash

# Root directory for PCA analysis
PCA_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/PCA_analysis"

# Script path for PCA processing
PCA_SCRIPT="/mnt/Cromosoma/mavila/backup/lcarrillo/scripts/make_evec_per_sample_hg38_lau_mD2_filters.Q20_v2.2.sh"

# Reference panel file
REFERENCE_FILE="1000GP.Phase3.v5a.MEGA.sites.10_pops.filt_hg38_sites_flipped_NO-alt_geno"

# Ensure PCA script exists
if [ ! -f "$PCA_SCRIPT" ]; then
    echo "PCA script not found: $PCA_SCRIPT"
    exit 1
fi

# Change to PCA directory
cd "$PCA_DIR" || exit

# Iterate over each individual directory
for DIR in *_PCA/; do
    # Define individual directory
    INDIVIDUAL_DIR="$PCA_DIR/$DIR"

    # Change to individual directory
    cd "$INDIVIDUAL_DIR" || continue

    # Find the BAM file with the expected suffix
    BAMFILE=$(ls *pmd3.trimmed9.MD.bam 2>/dev/null)

    # Check if BAM file exists
    if [ -z "$BAMFILE" ]; then
        echo "No BAM file found in $INDIVIDUAL_DIR"
        cd ..
        continue
    fi

    # Submit the job using qsub
    echo "Submitting job for $BAMFILE in $INDIVIDUAL_DIR"
    qsub "$PCA_SCRIPT" "$BAMFILE" "$REFERENCE_FILE"

    # Return to PCA directory
    cd ..
    sleep 60
done

exit
```
Modify the `PCA_DIR` and the suffix in `BAMFILE` if necessary.

	$tped_hash{$key}=$_;		      ## And the value is the line itself

	$fields[0] =~ s/chr//g;                   ## The hash built in the first loop is queried and the SNP coordinates (the map file line)

#!/bin/bash

# Root directory for PCA analysis
PCA_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/PCA_analysis"

# Directory to store symbolic links
PLOT_DIR="$PCA_DIR/plot_all"

# Create the plot_all directory if it doesn't exist
mkdir -p "$PLOT_DIR"

# Ensure the SNP summary file exists
if [ ! -f "$SNP_SUMMARY_FILE" ]; then
    echo "SNP summary file not found: $SNP_SUMMARY_FILE"
    exit 1

while IFS=$'\t' read -r INDIVIDUAL SNP_COUNT; do
    # Skip the header line
    if [ "$INDIVIDUAL" == "Individual" ]; then
        continue
    fi

    if [ "$SNP_COUNT" -ge 5000 ]; then
        # Define the directory for the individual
        INDIVIDUAL_DIR="$PCA_DIR/${INDIVIDUAL}_PCA"

        # Define the target files
        PED_FILE="$INDIVIDUAL_DIR"/*.filt_sites.ped

        if [ -f $PED_FILE ] && [ -f $MAP_FILE ]; then

            ln -sf "$MAP_FILE" "$PLOT_DIR/"
            echo "Links created for $INDIVIDUAL"
        else
            echo "PED or MAP file not found for $INDIVIDUAL"
        fi
    fi
done < "$SNP_SUMMARY_FILE"

for f in [0-9]*-*_*.{map,ped}; do # this, as it's written, only works with HSJN individuals

### 24.20 Generate filemap file
I use the next script to generate the filemap file needed for pong:
```
#!/bin/bash

# Define directory with Q files
QDIR="/Users/lcarrillo/Documents/Doctorado/30.Admixture/q_files"

# Output file name
OUTFILE="pong_filemap.txt"

# Clean the output file if already exist
> "$OUTFILE"

# Iterate over Q files in work directory
for file in "$QDIR"/*.Q; do
    # Extract file name
    filename=$(basename "$file")

    # Extract K and rep using regular expressions
    if [[ $filename =~ K([0-9]+)\.rep\.([0-9]+)\. ]]; then
        K=${BASH_REMATCH[1]}
        REP=${BASH_REMATCH[2]}
        echo "k${K}r${REP}	${K}	$file" >> "$OUTFILE"
    fi
done

echo "Generated file: $OUTFILE"
```
