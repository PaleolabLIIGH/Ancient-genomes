### 24.2 Prepare ancient genomes for PCA

```
)

# Ensure PCA directory exists
mkdir -p "$PCA_DIR"

# Change to the trimmed BAM files directory
cd "$TRIMMED_DIR" || exit

# Iterate over each trimmed BAM file
for BAMFILE in *pmd3.trimmed9.MD.bam; do
    # Check if BAM file exists
    if [ ! -f "$BAMFILE" ]; then
        echo "BAM file not found: $BAMFILE"
        continue
    fi

    # Check if index file exists
    BAI_FILE="$BAMFILE.bai"
    if [ ! -f "$BAI_FILE" ]; then
        echo "Index file not found for: $BAMFILE"
        continue
    fi

    # Extract individual ID from the BAM file name (assuming prefix format like "001-H089")
    INDIVIDUAL_ID=$(echo "$BAMFILE" | cut -d'_' -f1)
    INDIVIDUAL_DIR="$PCA_DIR/${INDIVIDUAL_ID}_PCA"

    # Create individual PCA directory
    mkdir -p "$INDIVIDUAL_DIR"

    # Create symbolic links for BAM and BAI files
    ln -sf "$TRIMMED_DIR/$BAMFILE" "$INDIVIDUAL_DIR/$BAMFILE"
    ln -sf "$TRIMMED_DIR/$BAI_FILE" "$INDIVIDUAL_DIR/$BAI_FILE"

    # Create symbolic links for reference panel files
    for REF_FILE in "${REFERENCE_FILES[@]}"; do
        ln -sf "$REFERENCE_PANEL_DIR/$REF_FILE" "$INDIVIDUAL_DIR/$REF_FILE"
    done
done

exit
```

```
#!/bin/bash

# 3 Arguments expected: $PCA_DIR, $REFERENCE_DIR, $BAMPATTERN 

# Root directory for PCA analysis
PCA_DIR="$1" # /mnt/data/mavila/mjimenezs/HC/pca_indigenas for example

# Script path for PCA processing ---> This works with the desired arguments. Thus, need no changes at all
PCA_SCRIPT="/mnt/data/mavila/mjimenezs/HC/scripts/make_evec_per_sample_hg38_mjimenez_mD2_filters.Q20_v2.2.sh"

# Reference panel file ---> Changed depending on whether you're working with 1KG or MXB
REFERENCE_FILE="$2" # The second argument --> On this case, PCA_MXB_1KG's refrence panel is used

# Ensure PCA script exists
if [ ! -f "$PCA_SCRIPT" ]; then
    echo "PCA script not found: $PCA_SCRIPT"
    exit 1
fi

# Change to PCA directory
cd "$PCA_DIR" || exit

# Iterate over each individual directory
for DIR in *_PCA/; do # Assuming they all have that suffix 
    # Define individual directory
    INDIVIDUAL_DIR="$PCA_DIR/$DIR"

    # Change to individual directory
    cd "$INDIVIDUAL_DIR" || continue

	BAMPATTERN="$3" # '"*pmd3.trimmed9.MD.bam"' --> You should modify the pattern if your filters were different 
    # Find the BAM file with the expected suffix
    BAMFILE=$(ls $BAMPATTERN 2>/dev/null)

    # Check if BAM file exists
    if [ -z "$BAMFILE" ]; then
        echo "No BAM file found in $INDIVIDUAL_DIR"
        cd ..
        continue
    fi

    # Submit the job using sbatch (SLURM) instead of qsub (SGE)
    echo "Submitting SLURM job for $BAMFILE in $INDIVIDUAL_DIR"
    sbatch << EOF
#!/bin/bash
#SBATCH --job-name=pca_${BAMFILE%.bam}
#SBATCH --output=pca_${BAMFILE%.bam}_%j.out 
#SBATCH --error=pca_${BAMFILE%.bam}_%j.err
#SBATCH --partition=defq
#SBATCH --time=24:00:00
#SBATCH --cpus-per-task=12
#SBATCH --mem=8G
#SBATCH --ntasks=1

# Run the PCA script with same arguments
"$PCA_SCRIPT" "$BAMFILE" "$REFERENCE_FILE"
EOF

    # Return to PCA directory
    cd ..
    sleep 60
done

exit

```

```
	chomp($_);

	$key=$fields[0] ."_".$fields[3];      ## Make a hash where the key is the chr and the position coordinates 
	$tped_hash{$key}=$_;		      ## And the value is the line itself

my $chr;

while(<MPILEUP>)

	chomp($_);
	@fields=split(/\t/,$_);                   ## The mpileup file is scanned the key is built with the chromosome and position coordinates
	$fields[0] =~ s/chr//g;                   ## The hash built in the first loop is queried and the SNP coordinates (the map file line)
```

```
#!/bin/bash

# Root directory for PCA analysis
PCA_DIR="$1" # /panfs/pan5/mavila/mjimenezs/HC/PCA_analysis
```

```
while IFS=$'\t' read -r INDIVIDUAL SNP_COUNT; do

    if [ "$INDIVIDUAL" == "Individual" ]; then
        continue
    fi

    # Check if SNP count is the number specified or more
    if [ "$SNP_COUNT" -ge "$MAX_SNPS" ]; then
        # Define the directory for the individual
        INDIVIDUAL_DIR="$PCA_DIR/${INDIVIDUAL}_PCA"

        # Define the target files
        PED_FILE=("$INDIVIDUAL_DIR"/$PATTERN.ped)
                MAP_FILE=("$INDIVIDUAL_DIR"/$PATTERN.map)

        if [ -f "$PED_FILE" ] && [ -f "$MAP_FILE" ]; then
            ln -sf "${PED_FILE[0]}" "$PLOT_DIR/"
                        ln -sf "${MAP_FILE[0]}" "$PLOT_DIR/"
            echo "Links created for $INDIVIDUAL"
        else
            echo "PED or MAP file not found for $INDIVIDUAL"
        fi
    fi
```

```
for f in [0-9]*-*_*.{map,ped}; do # this, as it's written, only works with HSJN individuals
```
