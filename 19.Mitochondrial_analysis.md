Before running the script, create a folder for each individual as follows:
```
#Create a folder for each individual and move the corresponding files
for file in *.bam; do

  #Obtain the individual's identifier (for example, '001-H089' from '001-H089_L001_R1-2_001.fastq.collapsed.bz2.realign.rCRS.MD.bam')
  individual=$(echo $file | cut -d'_' -f1)

  # Create the folder if it doesn't exist  
  mkdir -p $individual

  # Move the BAM file  
  mv $file $individual/

  # Move the BAI file (assuming the BAI file has the same prefix as the BAM file)  
  mv ${file}.bai $individual/
done
ls

This is the content in the script:
```
#!/bin/bash

# Root directory
ROOT_DIR="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/Mapping_mt/Schmutzi"

# qsub script
QSUB_SCRIPT="/mnt/Cromosoma/mavila/backup/lcarrillo/scripts/schmutzi_haplogrep3.sh"
REFERENCE_FILE="/mnt/Cromosoma/mavila/vvilla/rCRS/rCRS.fasta"

# Change to the root directory
cd "$ROOT_DIR" || exit

# Iterate over each subdirectory
for SUBDIR in */; do
    # Remove the trailing slash from the subdirectory name
    SUBDIR=${SUBDIR%/}

    # Change to the subdirectory
    cd "$SUBDIR" || continue

    # Find the BAM file in the subdirectory
    BAMFILE=$(ls *.bam 2>/dev/null)

    # Check if a BAM file was found
    if [ -z "$BAMFILE" ]; then
        echo "No BAM file found in $SUBDIR"
        cd ..
        continue
    fi

    # Submit the job to qsub
    qsub "$QSUB_SCRIPT" "$BAMFILE" "$REFERENCE_FILE"

    # Return to the root directory
    cd ..

    # Wait for the current job to finish before continuing
    # You can adjust this time according to how long each job usually takes
    sleep 60
done

exit

```

```
cd ..
for file in $(find /mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/Mapping_mt/Schmutzi/*H*/ -type f -name "*final.cont.est*"); do
   echo -n "$(basename $file | cut -d "_" -f 1)";
   more $file | awk '{ printf "\t%s", $0 } END { printf "\n" }';
done
