### 4.1 Understanding the script
Once you have your collapsed sequences, proceed to human mapping. However, before starting any process, you should carefully study what each command and program used in the script does. 

First visit this pages and learn about the programs used:
```

Here is the content for you to study:

```
#!/bin/bash
#Written by Maria Avila
# Modified by Viridiana Villa to Work at LIIGH
# Modified by Walter Ortega to work in FENIX
# Modofoed by Itzy Perez to obtain the unmapped fraction as fastq
# Modified by Laura Carrillo to work in parallel

#SBATCH --job-name=Mapping_Hg38_parallel
#SBATCH --output=Mapping_Hg38_parallel.%j.out
#SBATCH --error=Mapping_Hg38_parallel.%j.err
#SBATCH --cpus-per-task=24
#SBATCH --mem=32G
#SBATCH --nodes=1

set -euo pipefail

# =========================
# Environment
# =========================
source /etc/profile.d/modules.sh

module load htslib/1.2.1
module load gcc/5.1.0
module load bwa/0.7.13
module load samtools/1.6
module load picard/2.6.0
module load gatk/3.4-46
module load fastqc/0.11.3
module load bedtools/2.26

# =========================
# Arguments
# =========================
# $1 = reference genome (FASTA)
REF=$1

if [[ ! -f "$REF" ]]; then
    echo "ERROR: Reference genome not found: $REF"
    exit 1
fi

# =========================
# Sanity checks
# =========================
command -v bwa >/dev/null || { echo "bwa not found"; exit 1; }
command -v samtools >/dev/null || { echo "samtools not found"; exit 1; }
command -v GenomeAnalysisTK >/dev/null || { echo "GATK not found"; exit 1; }
command -v picard >/dev/null || { echo "Picard not found"; exit 1; }

# =========================
# Parallel configuration
# =========================
THREADS_PER_SAMPLE=12
MAX_JOBS=$(($SLURM_CPUS_PER_TASK / THREADS_PER_SAMPLE))

echo "=============================="
echo "CPUs available: $SLURM_CPUS_PER_TASK"
echo "Threads per sample: $THREADS_PER_SAMPLE"
echo "Samples running in parallel: $MAX_JOBS"
echo "=============================="

# =========================
# Create necessary directories
# =========================
mkdir -p Mapping_mt Unmapped

# =========================
# Main parallel loop
# =========================
for file in *.bz2
do
    (
    base=$(basename "$file" .bz2)
    base3=$(basename "$file" .collapsed.bz2)

    echo "=============================="
    echo "Processing sample: $base"
    echo "Started at: $(date)"
    echo "=============================="

    # =========================
    # Mapping
    # =========================
    bwa aln -l 500 -t $THREADS_PER_SAMPLE "$REF" <(bzcat "$base.bz2") | \
    bwa samse "$REF" \
        -r "@RG\tID:$base\tSM:$base" \
        - <(bzcat "$base.bz2") | \
    samtools view -bSh - > "$base.bwa.all.bam"

    samtools view -c "$base.bwa.all.bam" >> "${base}_REPORT.txt"

    samtools view -bh -F4 -q 30 "$base.bwa.all.bam" > "$base.bwa.bam"
    samtools view -bh -f 4 "$base.bwa.all.bam" > "$base.unmapped.bam"

    # =========================
    # Sort
    # =========================
    samtools sort -o "$base.bwa.sort.bam" "$base.bwa.bam"
    samtools view -c "$base.bwa.sort.bam" >> "${base}_REPORT.txt"

    rm "$base.bwa.all.bam" "$base.bwa.bam"

    # =========================
    # Remove PCR duplicates
    # =========================
    samtools rmdup -s "$base.bwa.sort.bam" "$base.bwa.sort.rmdup.bam"
    samtools view -c "$base.bwa.sort.rmdup.bam" >> "${base}_REPORT.txt"

    # =========================
    # Keep uniquely mapped
    # =========================
    samtools view -h "$base.bwa.sort.rmdup.bam" | \
        grep -v 'XT:A:R' | grep -v 'XA:Z' | \
        awk '{if($0~/X1:i:0/||$0~/^@/)print $0}' | \
        samtools view -bS - > "$base.bwa.sort.rmdup.uniq.bam"

    samtools view -c "$base.bwa.sort.rmdup.uniq.bam" >> "${base}_REPORT.txt"

    rm "$base.bwa.sort.rmdup.bam"

    # =========================
    # FastQC
    # =========================
    fastqc -f bam "$base.bwa.sort.rmdup.uniq.bam" &

    # =========================
    # Coverage stats
    # =========================
    samtools mpileup -f "$REF" -C 50 "$base.bwa.sort.rmdup.uniq.bam" | \
        tee "$base.bwa.sort.rmdup.uniq.pileup" | \
        awk '{if($3!="*"){sum+=$4}}END{print sum/NR "\n" NR}' \
        >> "${base}_REPORT.txt"

    samtools view "$base.bwa.sort.rmdup.uniq.bam" | \
        awk '{SUM+=length($10);DIV++}END{print SUM/DIV}' \
        >> "${base}_REPORT.txt"

    samtools index "$base.bwa.sort.rmdup.uniq.bam"

    # =========================
    # Add / Replace Read Groups
    # =========================
    picard AddOrReplaceReadGroups \
        I="$base.bwa.sort.rmdup.uniq.bam" \
        O="$base.bwa.sort.rmdup.uniq.RG.bam" \
        RGLB="$base3" \
        RGPL=illumina \
        RGPU="$base3" \
        RGSM="$base3"

    samtools index "$base.bwa.sort.rmdup.uniq.RG.bam"

    # =========================
    # GATK Realignment
    # =========================
    GenomeAnalysisTK \
        -T RealignerTargetCreator \
        -R "$REF" \
        -I "$base.bwa.sort.rmdup.uniq.RG.bam" \
        -o "$base.intervals"

    GenomeAnalysisTK \
        -T IndelRealigner \
        -R "$REF" \
        -I "$base.bwa.sort.rmdup.uniq.RG.bam" \
        -targetIntervals "$base.intervals" \
        -o "$base3.realign.bam"

    # =========================
    # MD tagging
    # =========================
    samtools calmd -b "$base3.realign.bam" "$REF" > "$base3.realign.MD.bam"
    samtools index "$base3.realign.MD.bam"

    tr '\n' ',' < "${base}_REPORT.txt" > "${base}_REPORT.csv"

    # =========================
    # MT-only BAM
    # =========================
    samtools view -h "$base3.realign.MD.bam" MT | \
        egrep 'MT|^@PG|^@H' | \
        samtools view -hbS - > "$base3.realign.rCRS.MD.bam"

    samtools index "$base3.realign.rCRS.MD.bam"
    samtools view -c "$base3.realign.rCRS.MD.bam" > "${base}_REPORT_mt.txt"
    tr '\n' ',' < "${base}_REPORT_mt.txt" > "${base}_REPORT_mt.csv"

    # Move MT files
    mv *rCRS* Mapping_mt/
    mv "${base}_REPORT_mt.txt" Mapping_mt/
    mv "${base}_REPORT_mt.csv" Mapping_mt/

    # =========================
    # Unmapped reads
    # =========================
    bedtools bamtofastq -i "$base.unmapped.bam" -fq "$base.unmapped.fastq"
    rm "$base.unmapped.bam"
    bzip2 "$base.unmapped.fastq"
    mv "$base.unmapped.fastq.bz2" Unmapped/

    # =========================
    # Schmutzi / Contammix prep
    # =========================
    cd Mapping_mt
    mkdir -p Schmutzi Contammix
    cp *rCRS*.ba* Schmutzi/
    cp *rCRS*.ba* Contammix/
    cd ..

    echo "=============================="
    echo "Finished sample: $base at $(date)"
    echo "=============================="

    ) &

    # Control number of parallel jobs
    if [[ $(jobs -r -p | wc -l) -ge $MAX_JOBS ]]; then
        wait -n
    fi

done

# Wait for all remaining jobs to finish
wait

# Wait for any background FastQC processes
wait

echo "=============================="
echo "All samples processed successfully at $(date)"
echo "=============================="
```

Next, go to your working directory and create a folder for mapping where you will place symbolic links to the collapsed files:

```

```

```

NOTE 2: If the links appear in red, they were not copied correctly and the process needs to be repeated.

NOTE 3: If the links appear in dark blue instead of light blue, symbolic hard links were created instead of soft links for some reason. In this case, do not delete them because deleting them will also 
remove your original file. Just continue with the process.

As you will see, a series of files have been generated for each individual, but this is quite disorganized. Therefore, we will now arrange the results into 
folders by individual, which will be useful for future analyses.

We will use the regex expressions described in the file for renaming to organize the files. Remember to adapt the regex according to the structure of your 
files:

```
for file in *; do
    if [[ -f "$file" ]]; then
        individual=$(echo "$file" | grep -Eo "^[0-9]+-[A-Z]+[0-9]+") #Modify accordly to your regex
        if [[ -n "$individual" ]]; then
            mkdir -p "${individual}-bam" #Modify if you desire other name as -WISC-bam or other description
            if [[ "$file" == *".bz2."* || "$file" == *".bw"* || "$file" == *".i"* ]]; then
                mv "$file" "${individual}-bam/" #Modify if you change the previous name
            fi
        fi
    fi
done
ls
```

### Method 2
In this case the target is to remove human contamination, so we don't need all the steps in the previous script, therefore use this other one:
```
#!/bin/bash
### Written by Laura Carrillo to get only the unmmaped fraction
#SBATCH --job-name=Trim_map_script_v_parallel
#SBATCH --output=Trim_map_script_v_parallel.%j.out
#SBATCH --error=Trim_map_script_v_parallel.%j.err
#SBATCH --cpus-per-task=24
#SBATCH --mem=32G
#SBATCH --nodes=1

set -euo pipefail

# =========================
# Environment
# =========================
source /etc/profile.d/modules.sh

module load htslib/1.2.1
module load gcc/5.1.0
module load bwa/0.7.13
module load samtools/1.6
module load bedtools/2.26

# =========================
# Arguments
# =========================
# $1 = reference genome (FASTA)
REF=$1

if [[ ! -f "$REF" ]]; then
    echo "ERROR: Reference genome not found: $REF"
    exit 1
fi

# =========================
# Sanity checks
# =========================
command -v bwa >/dev/null || { echo "bwa not found"; exit 1; }
command -v samtools >/dev/null || { echo "samtools not found"; exit 1; }
command -v bedtools >/dev/null || { echo "bedtools not found"; exit 1; }

# =========================
# Parallel configuration
# =========================
THREADS_PER_SAMPLE=12
MAX_JOBS=$(($SLURM_CPUS_PER_TASK / THREADS_PER_SAMPLE))

echo "=============================="
echo "CPUs available: $SLURM_CPUS_PER_TASK"
echo "Threads per sample: $THREADS_PER_SAMPLE"
echo "Samples running in parallel: $MAX_JOBS"
echo "=============================="

# =========================
# Main parallel loop
# =========================
for file in *.bz2
do
    (
    base=$(basename "$file" .bz2)

    echo "=============================="
    echo "Processing sample: $base"
    echo "=============================="

    # =========================
    # Mapping
    # =========================
    bwa aln -l 500 -t $THREADS_PER_SAMPLE "$REF" <(bzcat "$file") | \
    bwa samse "$REF" \
        -r "@RG\tID:$base\tSM:$base" \
        - <(bzcat "$file") | \
    samtools view -bSh - > "$base.bwa.all.bam"

    # =========================
    # Counts
    # =========================
    TOTAL=$(samtools view -c "$base.bwa.all.bam")
    MAPPED=$(samtools view -c -F 4 "$base.bwa.all.bam")
    UNMAPPED=$(samtools view -c -f 4 "$base.bwa.all.bam")

    echo "$TOTAL" > "${base}_REPORT.txt"
    echo "$MAPPED" >> "${base}_REPORT.txt"
    echo "$UNMAPPED" >> "${base}_REPORT.txt"

    tr '\n' ',' < "${base}_REPORT.txt" > "${base}_REPORT.csv"

    # =========================
    # Extract unmapped reads
    # =========================
    samtools view -bh -f 4 "$base.bwa.all.bam" > "$base.unmapped.bam"

    # Convert to FASTQ
    bedtools bamtofastq \
        -i "$base.unmapped.bam" \
        -fq "$base.unmapped.fastq"

    bzip2 -9 "$base.unmapped.fastq"

    mkdir -p Unmapped
    mv "$base.unmapped.fastq.bz2" Unmapped/

    # Cleanup
    rm "$base.bwa.all.bam"
    rm "$base.unmapped.bam"

    echo "Finished sample: $base"
    
    ) &

    # Control number of parallel jobs
    if [[ $(jobs -r -p | wc -l) -ge $MAX_JOBS ]]; then
        wait -n
    fi

done

# Wait for all remaining jobs to finish
wait

echo "=============================="
echo "All samples processed"
echo "=============================="
```
