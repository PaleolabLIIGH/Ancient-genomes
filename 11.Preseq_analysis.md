Perform a loop through all files ending with `*sort.bam`:

```
### Written by Rigoberto Padilla
### Modified by Laura Carrillo Olivas to run it in screen instead of a job
for file in *sort.bam; do
  #Check if the file exists:
  if [ -f "$file" ]; then
    #Extract the file prefix (filename without the .bam extension):
    prefix=$(basename "$file" .bam)
    # Execute preseq lc_extrap and save the output
    preseq lc_extrap -B -o "${prefix}_yield_estimates.txt" "$file"
    # Execute preseq c_curve and save the output
    preseq c_curve -B -o "${prefix}_complexity_output.txt" "$file"
    # Execute bam2mr and save the output
    bam2mr "$file" -o "${prefix}.mr" -L300 -R 25000000
    # Sort the output of bam2mr
    sort -k 1,1 -k2,2n -k3,3n "${prefix}.mr" > "${prefix}.mr.sort"
    # Execute preseq gc_extrap and save the output
    preseq gc_extrap "${prefix}.mr.sort" -o "${prefix}_future_coverage.txt"
    #Print a message indicating the file has been processed:
    echo "Processed: $file"
  fi
done
```
