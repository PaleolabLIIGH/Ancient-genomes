### 10.4 Plot the Edit distance, the breath and depth coverage and the quality
Open RStudio
Get the packages if necessary

```
#Written by Nikolay Oskolkov 
#Evaluating Breadth and Evenness of Coverage

### Option 1
# Load necessary libraries
library("Rsamtools")
library(GenomicAlignments)
# Path to your BAM file
bam_path <- "~/Documents/Doctorado/15.bams/genomes/capturas/salmonella/021-H238-WISC_L001_R1-2_001.fastq.collapsed.unmapped.fastq.bwa.sort.rmdup.uniq.bam"

# Read the BAM file
bam <- readGAlignments(bam_path)

# Get the start positions of reads
start_positions <- start(bam)

# Create a histogram of start positions
histogram <- hist(start_positions, breaks = 100, plot = FALSE)

# Plot a line graph showing the distribution of start positions
plot(histogram$mids, histogram$counts, type = "l", 
     xlab = "Start Position", ylab = "Number of Reads",
     main = "Read Distribution Across the Genome")

### Option 2
# Read output from samtools depth command
df <- read.delim("~/Documents/Doctorado/15.bams/genomes/capturas/salmonella/021-H238-WISC_L001_R1-2_001.fastq.collapsed.unmapped.fastq.bwa.sort.rmdup.uniq.boc", header = FALSE, sep = "\t")
names(df) <- c("Ref", "Pos", "N_reads")

# Split reference genome in tiles, compute breadth of coverage for each tile
N_tiles <- 500
step <- (max(df$Pos) - min(df$Pos)) / N_tiles
tiles <- c(0:N_tiles) * step; boc <- vector()
for(i in 1:length(tiles))
{
  df_loc <- df[df$Pos >= tiles[i] & df$Pos < tiles[i+1], ]
  boc <- append(boc, rep(sum(df_loc$N_reads > 0) / length(df_loc$N_reads),
                         dim(df_loc)[1]))
}

boc[is.na(boc)]<-0; df$boc <- boc
plot(df$boc ~ df$Pos, type = "s", xlab = "Genome position", ylab = "Coverage")
abline(h = 0, col = "red", lty = 2)
mtext(paste0(round((sum(df$N_reads > 0) / length(df$N_reads)) * 100, 2), 
             "% of genome covered"), cex = 2)

### Assessing Mapping Quality
hist(as.numeric(readLines("~/Documents/Doctorado/15.bams/genomes/capturas/salmonella/021-H238-WISC_L001_R1-2_001.fastq.collapsed.unmapped.fastq.bwa.sort.rmdup.uniq.bam_mapq.txt")), col = "darkred", breaks = 100)

### Analyzing Edit Distance
param <- ScanBamParam(tag = "NM")
bam <- scanBam("~/Documents/Doctorado/15.bams/genomes/capturas/salmonella/021-H238-WISC_L001_R1-2_001.fastq.collapsed.unmapped.fastq.bwa.sort.rmdup.uniq.bam", param = param)
# Define bar color as green
green_color <- "#01796F"  # Example color, adjust to your preference

barplot(table(bam[[1]]$tag$NM), ylab = "Number of Reads", xlab = "Number of Mismatches", col = colors)
title("Edit Distance")
```
