import click

@click.command()
@click.option('--filename', help='name of input file')


def likelihoods(filename):

    f = open('Likelihoods_order.txt', 'a')
    file = open(filename).readlines()
    f.write('File_name'+ ',' + 'Loglikelihood' + '\n')
    for line in file:
        outfile= line.strip()
        insideOUTfile = open(outfile).readlines()
        for linea in insideOUTfile:

            if linea.startswith('Loglikelihood'):
                likeli = linea.split(':')
                likelihood = float(likeli[1].strip())
                f.write(str(outfile)+ ',' + str(likelihood)+ '\n')
    f.close()

likelihoods()
