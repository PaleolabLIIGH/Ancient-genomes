### 3.1 Get the script and modify it
Use the following script to remove the adapters:
Modify only your email, your path (wdir) and the variables fbz1 and fbz2

Go to your main work directory and create a new directory called scripts

```
cd /mnt/Cromosoma/mavila/lcarrillo/
mkdir scripts
```
Copy the next script 

```
cd scripts
cp /mnt/Cromosoma/mavila/backup/lcarrillo/scripts/Adapter_V2_pair_end_lco_corrected.sh .
```
Study the code and make sure to change the email, tha path or wdir and if necessary fbz1 and fbz2 (only if your files are diferents)
The content is here:

```
### Written by Viridiana Villa
### Modified by Laura Carrillo to do a loop and create folders with better order
### adapterremoval for all samples and lanes
### for samples with L001_R1 and L001_R2 only
#Your job name
#$-N AdapterremovalLCO
#Use current working directory
#$-cwd

#Join stdout and stderr
#$-j y
# Run job through bash shell
#$-S /bin/bash

#Send an email after the job has finished
#$-m e
#$-M laura-carrillo-olivas@hotmail.com


#If modules are needed, source modules environment (Do not delete the next line):
 . /etc/profile.d/modules.sh


module load adapterremoval/1.5.4

#################
#################
#################
###The first argument is the name of one of the fastq files *L001_R1_001.fastq.gz
# Commands in the next lines:

#work directory as a variable:
wdir=$"/mnt/Cromosoma/mavila/lcarrillo/capturas_salmonella_myco"

###
###
###

#Set files as variables:
fbz1=$"_L001-4_R1_001.fastq.bz2"
fbz2=$"_L001-4_R2_001.fastq.bz2"

#Loop to adapter removal
for file in $(find $wdir -type f -name "*$fbz1*"); do
    nm=$(basename "$file" | cut -d'_' -f1)
    echo $nm
    rem1=$(echo "${nm}${fbz1}")
    echo $rem1
    rem2=$(echo "${nm}${fbz2}")
	echo $rem2
	reads=$"_L001_R1-2_001.fastq"
	fq3=$(echo "${nm}${reads}")
	echo $fq3

	/mnt/Cromosoma/mavila/vvilla/scripts/AdapterRemoval2/adapterremoval-2.3.1/build/AdapterRemoval --file1 $rem1 --file2 $rem2 --basename $fq3 --trimns --trimqualities --qualitybase 33 --minlength 30 --collapse --bzip2


	word=$"_adapter_collapsed.txt"
	report=$(echo "${nm}${word}")
	echo $report

	echo "Reads" > $report

	bz3=$"_L001-4_R1_001.fastq.bz2"
	fq4=$(echo "${nm}${bz3}")
	echo $fq4

	bzcat $fq4 | wc -l >> $report

	echo "Collapsed after adapterremoval" >> $report

	coll=$(echo "${fq3}.collapsed.bz2")
	echo $coll
	colltrun=$(echo "${fq3}.collapsed.truncated.bz2")
	echo $colltrun
	disc=$(echo "${fq3}.discarded.bz2")
	echo $disc
	pair1=$(echo "${fq3}.pair1.truncated.bz2")
	echo $pair1
	pair2=$(echo "${fq3}.pair2.truncated.bz2")
	echo $pair2
	sett=$(echo "${fq3}.settings")
	echo $sett
	sing=$(echo "${fq3}.singleton.truncated.bz2")
	echo $sing

	bzcat $coll | wc -l >> $report

	fastq_dir=$(echo "${nm}_fastq")
	echo $fastq_dir

	mkdir $fastq_dir
	mv $coll $fastq_dir/
	mv $colltrun $fastq_dir/
	mv $disc $fastq_dir/
	mv $pair1 $fastq_dir/
	mv $pair2 $fastq_dir/
	mv $sett $fastq_dir/
	mv $sing $fastq_dir/

done


exit;
```
