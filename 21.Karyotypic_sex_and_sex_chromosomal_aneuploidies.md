### 21.3 PMD bam files
The important file is the `*.realign.MD.bam`. However, since this is a very delicate analysis, it is better to remove modern contamination at once using pmdtools and then compare the results of
the original with each of the levels as describen previosly. Therefore, create the necessary pmd*bam files.
```
qlogin -pe openmp 12
cd /mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/
# Working directory
wdir="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/"

# Loop to process BAM files
for file in $(find $wdir/*H*-bam/ -type f -name "*.realign.MD.bam"); do
    module load htslib/1.2.1
    module load python26/2.6.9
    module load samtools/1.2
    
    # Process the file with pmdtools for each PMD level
    for level in 0 1 2 3; do
        samtools view -h -q 30 $file | python /mnt/Cromosoma/mavila/backup/lcarrillo/scripts/pmdtools.py --header --threshold $level | samtools view -Sb - > "${file%.MD.bam}.pmd${level}.MD.bam"
        echo "Processed file: $file with PMD level $level"
    done
done
exit
```

### 21.5 Sex determination of the pmd*bam files
```
qlogin -pe openmp 12
cd /mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/
# Path to your script
script_path="/mnt/Cromosoma/mavila/lcarrillo/scripts/X_Y_determination_lco_2024.sh"
# Path where all your BAM folders are located
main_folder="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38"

# Loop through each subfolder
for SUBFOLDER in $main_folder/*-bam; do
    # Loop through each PMD level
    for pmd_level in pmd0 pmd1 pmd2 pmd3; do
        # Construct the BAM file path for the current PMD level
        BAM_FILE=$SUBFOLDER/*bz2.realign.${pmd_level}.MD.bam
        
        # Print variables for debugging
        echo "SUBFOLDER: $SUBFOLDER"
        echo "BAM_FILE: $BAM_FILE"
        
        # Check if the BAM file exists
        if ls $BAM_FILE 1> /dev/null 2>&1; then
            # Get the folder name
            FOLDER_NAME=$(basename "$SUBFOLDER")
            # Execute the script for the BAM file
            sh "$script_path" "$BAM_FILE" > "$main_folder/$FOLDER_NAME/${FOLDER_NAME}_sex_determination_${pmd_level}.txt"
            echo "Processed: $FOLDER_NAME with PMD level $pmd_level"
        else
            echo "No BAM file found in subfolder: $SUBFOLDER for PMD level $pmd_level"
        fi
    done
done
```
