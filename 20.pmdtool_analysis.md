### 20.2 Process the bam files for each PMD level
```
qlogin -pe openmp 12
cd /mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/Mapping_mt/Schmutzi/
# Working directory
wdir="/mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/Mapping_mt/Schmutzi"

# Loop to process BAM files
for file in $(find $wdir/*H*/ -type f -name "*rCRS.MD.bam"); do
    module load htslib/1.2.1
    module load python26/2.6.9
    module load samtools/1.2
    
    # Process the file with pmdtools for each PMD level
    for level in 0 1 2 3; do
        samtools view -h -q 30 $file | python /mnt/Cromosoma/mavila/backup/lcarrillo/scripts/pmdtools.py --header --threshold $level | samtools view -Sb - > "${file%.MD.bam}.pmd${level}.MD.bam"
        echo "Processed file: $file with PMD level $level"
    done
done
exit
```

In the case of Contammix, you could use the next code for each pmd analysis:
```
cd /mnt/Cromosoma/mavila/lcarrillo/HSJN-concatenado/Mapping_hg38/Mapping_mt/Contammix
for subfolder in *-H*; do
    # Check if the pmd0 subfolder exists
    if [ -d "$subfolder/pmd0" ]; then
        # Extract the folder name
        folder=$(basename "$subfolder")

        # Use find to search for the file in the pmd0 subfolder
        file=$(find "$subfolder/pmd0" -type f -name "${folder}*pmd0*_plus311.fa.contamix.out" 2>/dev/null)

        # Check if find found the file
        if [ -n "$file" ]; then
            # Extract the columns from the .out file
            columns=$(awk '{ printf "\t%s", $0 } END { if (NR == 0) printf "NA\tNA\tNA\tNA\tNA\tNA"; printf "\n" }' "$file")
            # Print the row with tab delimiter
            echo -e "$folder$columns"
        else
            # Print a row with NA values if the file is not found
            echo -e "$folder\tNA\tNA\tNA\tNA\tNA\tNA"
        fi
    else
        # Print a row with NA values if the pmd0 subfolder does not exist
        echo -e "$folder\tNA\tNA\tNA\tNA\tNA\tNA"
    fi
done
```
