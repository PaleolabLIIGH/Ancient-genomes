### 27.14 Organize genes by PAIs:
In this case Salmonella has PAIs, and I desire to analyze these genes with that organization, so I create this script in python:
```
# List of pathogenicity islands (start, end, name)
pathogenicity_islands = [
    (290850, 326375, "SPI-6"),  # First region of SPI-6
    (334900, 345883, "SPI-6"),  # Second region of SPI-6
    (1526737, 1546154, "SPI-12"),
    (2364388, 2406210, "SPI-2"),
    (2539963, 2548259, "SPI-11"),
    (2678960, 2699944, "SPI-5"),
    (2851591, 2860986, "SPI-9"),
    (2940644, 2984129, "SPI-1"),
    (3885729, 3899837, "SPI-3"),
    (4406116, 4434245, "SPI-4"),
    (4496873, 4578677, "SPI-7")
]

# Read the BED file of genes
bed_file = "salmonella_genes_virulencia.bed"
with open(bed_file, "r") as f:
    genes = [line.strip().split("\t") for line in f]

# Dictionary to store genes by island
genes_by_island = {island[2]: [] for island in pathogenicity_islands}
genes_by_island["no_island"] = []  # For genes that do not belong to any island

# Function to extract the short gene name from the BED line
def extract_gene_name(bed_line):
    # The gene name is in the 4th column, after "Name="
    attributes = bed_line[3].split(";")
    for attr in attributes:
        if attr.startswith("Name="):
            return attr.replace("Name=", "")
    return "unknown_gene"  # Fallback if no gene name is found

# Classify genes according to the islands
for gene in genes:
    chromosome, start, end, attributes = gene
    start = int(start)
    end = int(end)
    found = False

    # Extract the short gene name
    gene_name = extract_gene_name(gene)

    # Check if the gene is within any region of any island
    for island_start, island_end, island_name in pathogenicity_islands:
        if start >= island_start and end <= island_end:
            # Store only the chromosome, start, end, and short gene name
            genes_by_island[island_name].append([chromosome, start, end, gene_name])
            found = True

    # If not in any island, add it to "no_island"
    if not found:
        genes_by_island["no_island"].append([chromosome, start, end, gene_name])

# Save genes into separate BED files
for island, island_genes in genes_by_island.items():
    with open(f"{island}.bed", "w") as f:
        for gene in island_genes:
            f.write("\t".join(map(str, gene)) + "\n")

print("BED files generated by island with short gene names!")
```
