#!/bin/bash
# Written by Laura Carrillo Olivas
# Download GCF assemblies from NCBI RefSeq
# and produce a concatenated multifasta file

set -euo pipefail

# Input
ACCESSIONS_FILE="assemblies.txt"
SUMMARY_REFSEQ="assembly_summary_refseq.txt"
GENUS="Citrobacter" ### Modfied if needed

# Output
OUTDIR="genomes"
MULTIFASTA_DIR="multifasta"
MULTIFASTA_FILE="$MULTIFASTA_DIR/$GENUS.fasta"

mkdir -p "$OUTDIR" "$MULTIFASTA_DIR"
> "$MULTIFASTA_FILE"

# Download RefSeq assembly summary
if [ ! -f "$SUMMARY_REFSEQ" ]; then
    echo "Downloading RefSeq assembly_summary..."
    wget -q ftp://ftp.ncbi.nlm.nih.gov/genomes/refseq/bacteria/assembly_summary.txt \
        -O "$SUMMARY_REFSEQ"
fi

# Filter Genus + selected GCF accessions
grep -Ff "$ACCESSIONS_FILE" "$SUMMARY_REFSEQ" \
    | grep -i "$GENUS" \
    > filtered_assemblies.txt

# Download genomic FASTA and build multifasta
while read -r line; do
    FTP_URL=$(echo "$line" | cut -f20)   # ftp_path

    if [[ "$FTP_URL" == ftp://* || "$FTP_URL" == https://* ]]; then
        BASE=$(basename "$FTP_URL")
        FILE="$OUTDIR/${BASE}_genomic.fna.gz"

        echo "Downloading $BASE ..."
        wget -q -O "$FILE" "${FTP_URL}/${BASE}_genomic.fna.gz"

        if [ -s "$FILE" ]; then
            gunzip -c "$FILE" >> "$MULTIFASTA_FILE"
        else
            echo "WARNING: failed to download $BASE"
        fi
    fi
done < filtered_assemblies.txt

#!/bin/bash
# Written by Laura Carrillo Olivas
# Extract FASTA headers from all GCA_*/GCF_* genomic files and create a table with: AssemblyID | SequenceAccession | Description

output="accession_headers_table.tsv"
echo -e "Assembly_ID\tSequence_Accession\tDescription" > "$output"

# Process all genomic FASTA .gz files
for f in *_genomic.fna.gz; do
    assembly_id=$(basename "$f" | cut -d'_' -f1-2)   # e.g. GCA_901290765.1
    echo "Processing $assembly_id ..."

    # Extract headers inside the file
    zgrep "^>" "$f" | while IFS= read -r line; do
        seq_acc=$(echo "$line" | cut -d' ' -f1 | sed 's/^>//')
        desc=$(echo "$line" | cut -d' ' -f2-)
        echo -e "${assembly_id}\t${seq_acc}\t${desc}" >> "$output"
    done
done

echo "Table generated: $output"

# Written by Laura Carrillo Olivas
set -euo pipefail

INFO="citrobacter_genomas_completos.txt"
HEADERS="accession_headers_table.tsv"
OUT="accession_headers_with_strain.tsv"
awk -F'\t' -v OFS='\t' -v out="$OUT" '

# Read Genus metadata
NR==1 { next }
NR==FNR {
    gcf = $2        # Assembly Accession (GCF)
    strain = $5     # Organism Infraspecific Names Strain
    map[gcf] = strain
    next
}

# Process headers table
FNR==1 {
    print $0, "strain" > out
    next
}

{
    gcf = $1  #### Modify if needed
    if (gcf in map)
        print $0, map[gcf] > out
    else
        print $0, "" > out
}
' "$INFO" "$HEADERS"

echo "Output file: $OUT"

Now we can count the results with the next script:
```
/mnt/data/mavila/lcarrillo/scripts/count_by_gcf_competitive_mapping.sh
```
Go to your work directory, copy the script and run it. This is the script:
```
#!/bin/bash
# Written by Laura Carrillo Olivas

# Load the modules
module load samtools/1.16.1


# Input files
GCF_TABLE="accession_headers_with_strain.tsv"
BAM_DIR="$PWD"


# Output files
OUTFILE="read_counts_matrix.tsv"
TMPDIR="tmp_counts"
mkdir -p "$TMPDIR"


# Get BAM files and sample names
BAMS=($(find "$BAM_DIR" -type f -name "*uniq.bam"))
SAMPLES=()

for bam in "${BAMS[@]}"; do
    sample=$(basename "$bam" | cut -d "_" -f 1)
    SAMPLES+=("$sample")
done


# Write header
echo -ne "GCF_ID" > "$OUTFILE"
for s in "${SAMPLES[@]}"; do
    echo -ne "\t$s" >> "$OUTFILE"
done
echo >> "$OUTFILE"

# Extract unique GCF IDs
GCF_IDS=$(tail -n +2 "$GCF_TABLE" | cut -f1 | sort -u)

# Count reads per GCF
for gcf in $GCF_IDS; do
    echo "Processing $gcf ..."

    # Get contig accessions for this GCF
    accessions=$(awk -v id="$gcf" '$1 == id {print $2}' "$GCF_TABLE")

    # Build regex pattern (contig1|contig2|...)
    pattern=$(echo "$accessions" | paste -sd'|' -)

    # Start row
    echo -ne "$gcf" > "$TMPDIR/$gcf.tmp"

    # Count reads per BAM
    for bam in "${BAMS[@]}"; do
        count=$(samtools view "$bam" | grep -E "$pattern" | wc -l)
        echo -ne "\t$count" >> "$TMPDIR/$gcf.tmp"
    done
    echo >> "$TMPDIR/$gcf.tmp"
done

# Merge temp files
cat "$TMPDIR"/*.tmp >> "$OUTFILE"

echo "Read count matrix generated:"
echo "   $OUTFILE"
```
