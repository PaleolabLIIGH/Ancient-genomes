Use the next code to get the reads of each genome:
```
#Load the modules
module load htslib/1.2.1
module load gcc/5.1.0
module load bwa/0.7.13
module load samtools/1.16.1

# Read the list of accession numbers
accession_list="accesion_list.txt"

# Iterate over each accession number in the list
while IFS= read -r accession; do
  # Iterate over the BAM files
  for file in $(find /mnt/Cromosoma/mavila/lcarrillo/capturas_salmonella_myco/Mapping_hg38/Unmapped/salmonella_competitiva -type f -name "*uniq.bam*"); do 
    name=$(basename "$file" | cut -d "_" -f 1) # Extract the sample name from the file
    reads=$(samtools view "$file" | grep "$accession" | wc -l) # Count the reads that match the accession number
    echo "$accession $name $reads" # Output the accession number, sample name, and read count
  done
done < "$accession_list"
```
